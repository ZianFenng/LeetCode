        """
        :type nums: List[int]
        :type target: int
        :rtype: int
        """
        n = len(nums)
        if n < 3:
            return 0
        nums.sort()
        
        i = 0
        count = 0
        while i < n - 1:
            l = i + 1
            r = n - 1
            while l < r:
                if nums[i] + nums[l] + nums[r] >= target:
                    r -= 1
                else:
                    count += r - l
                    # r = n - 1
                    l += 1
            i += 1
        return count

class Solution(object):
    def threeSumClosest(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: int
        """
        n = len(nums)
        if n <= 3:
            return sum(nums)
        nums.sort()
        i = 0 
        dif = float("inf")
        clsTar = float("inf")
        while i < n-1:
            l = i + 1
            r = n - 1
            
            while l < r:
                s = nums[i] + nums[l] + nums[r]
                diff = s - target
                if abs(diff) < dif:
                    dif = abs(diff)
                    clsTar = s
                if diff < 0:
                    l += 1
                elif diff > 0:
                    r -= 1
                else:
                    return target
                                
            while i < n - 1 and nums[i+1] == nums[i]:
                i += 1
            i += 1
        return clsTar
